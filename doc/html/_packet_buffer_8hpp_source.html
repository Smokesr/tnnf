116  uint16_t packetSize = 0;

118  while(mCurrentlyStoredBytes >= sizeof(uint16_t)) { //if the buffer store enough bytes to get the size of the next packet
119  packetSize = ntohs( *((uint16_t*) cursor));
120 
121  if(mCurrentlyStoredBytes >= packetSize) { //if there is the full packet
122  std::stringstream ss;

124  ss << mBuffer[i];
125  }
126 
127  cursor = &mBuffer[sizeof(uint16_t)];
128  mStoredPackets.emplace(ntohs( *((uint16_t*) cursor)), ss.str()); //construct a new packet

131  mBuffer[i - packetSize] = mBuffer[i];
