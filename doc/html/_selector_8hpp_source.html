140  void update() {
141  if(mFdReadablePointer != nullptr || mFdWritablePointer != nullptr || mFdFaultyPointer != nullptr) {
142  mFdWritable = mFdSockets;
143  mFdReadable = mFdSockets;
144  mFdFaulty = mFdSockets;
145 
146  int selectError;
147  if((selectError = select(mSocketsMax+1, mFdReadablePointer, mFdWritablePointer, mFdFaultyPointer, mTimeout)) <= 0) {
148  clearTemp();
149  if(selectError == 0) {
150  gCommonErrorFunction(ERROR_SELECTOR_TIMEOUT, "Selector timed out.");
151  return;
152  }
153  else {
154  gCommonErrorFunction(ERROR_SELECTOR_FAIL, "Selector error.");
155  return;
156  }
157  }
158  else {

160  mWritable->clear();
161

164  mWritable->push_back(i);
